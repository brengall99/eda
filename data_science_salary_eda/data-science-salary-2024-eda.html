What are the geographical trends of the data?¶

print()

# Define mapping of country codes to country names
country_names = {
    'QA': 'Qatar',
    'IL': 'Israel',
    'MY': 'Malaysia',
    'PR': 'Puerto Rico',
    'US': 'United States',
    'NZ': 'New Zealand',
    'CA': 'Canada',
    'SA': 'Saudi Arabia',
    'CH': 'Switzerland',
    'EG': 'Egypt'
}

# Replace country codes with country names
df['employee_residence'] = df['employee_residence'].map(country_names)

# Group the data by country and calculate the mean salary for each country
mean_salary_by_country = df.groupby('employee_residence')['salary_in_usd'].mean().reset_index()

# Sort the data by mean salary in descending order
mean_salary_by_country_sorted = mean_salary_by_country.sort_values(by='salary_in_usd', ascending=False)

# Select the top 10 countries
top_10_countries = mean_salary_by_country_sorted.head(10)

# Plotting
plt.figure(figsize=(10, 6))
plt.bar(top_10_countries['employee_residence'], top_10_countries['salary_in_usd'], color='skyblue')
plt.xlabel('Country')
plt.ylabel('Mean Salary (USD)')
plt.title('Top 10 Countries by Mean Salary')
plt.xticks(rotation=45)  # Rotate x-axis labels for better readability
plt.grid(True)  # Add gridlines
plt.show()
